レシピ#6-7: コラボレーションでアプリを進化させる

06章で作った個人用アプリは「動く」状態です。しかし、実際に使い始めると、改善点や「こんな機能があったら」というアイデアが次々と浮かんできます。

一人で作り、みんなで育てる。

このレシピでは、コラボレーションを通じてアプリを進化させる方法を学びます。特に、特化型機能の追加は、BYOAの真価を発揮する重要な進化の手段です。

コラボレーションの3つのパターン

パターン1: セルフ改善 + AI

自分 + Claude で改善を重ねる

あなたが:

Claude に:

// 改善後（コラボで）
function saveTasks(tasks) {
  try {
    localStorage.setItem('tasks', JSON.stringify(tasks))
  } catch (error) {
    if (error.name === 'QuotaExceededError') {
      alert('保存容量が不足しています。古いタスクを削除してください。')
    } else {
      console.error('保存に失敗しました:', error)
      alert('データの保存に失敗しました。')
    }
  }
}

// 投資可能額の自動算出
function calculateInvestmentCapacity() {
  const monthlyData = getMonthlyFinancialData()

  return {
    income: monthlyData.totalIncome,
    fixedExpenses: monthlyData.fixedExpenses, // 家賃、光熱費等
    variableExpenses: monthlyData.variableExpenses, // 食費、交通費等
    emergencyFund: monthlyData.totalIncome * 0.2, // 緊急資金20%
    investableAmount: monthlyData.totalIncome
                    - monthlyData.fixedExpenses
                    - monthlyData.variableExpenses
                    - (monthlyData.totalIncome * 0.2),

    recommendation: generateInvestmentRecommendation()
  }
}

// リスク許容度診断
function assessRiskTolerance() {
  const data = getUserFinancialData()

  const stability = data.incomeStability // 収入の安定性
  const expenses = data.expenseRatio // 支出比率
  const emergency = data.emergencyFundMonths // 緊急資金（月数）

  if (emergency >= 6 && stability === 'high' && expenses < 0.7) {
    return {
      level: 'aggressive',
      suggestion: '株式 70% / 債券 20% / 現金 10%',
      reason: '収入が安定し、緊急資金も十分です'
    }
  } else if (emergency >= 3 && stability === 'medium') {
    return {
      level: 'balanced',
      suggestion: '株式 50% / 債券 30% / 現金 20%',
      reason: 'バランス型がおすすめです'
    }
  } else {
    return {
      level: 'conservative',
      suggestion: 'まずは緊急資金を6ヶ月分貯めましょう',
      reason: '投資の前に生活防衛資金を確保してください'
    }
  }
}

// モジュール化
const TodoApp = (() => {
  let tasks = []

  // 初期化
  function init() {
    loadTasks()
    renderTasks()
    attachEventListeners()
  }

  // LocalStorage からロード（エラーハンドリング付き）
  function loadTasks() {
    try {
      const stored = localStorage.getItem('tasks')
      tasks = stored ? JSON.parse(stored) : []
    } catch (error) {
      console.error('タスクの読み込みに失敗:', error)
      tasks = []
    }
  }

  // LocalStorage へ保存（エラーハンドリング付き）
  function saveTasks() {
    try {
      localStorage.setItem('tasks', JSON.stringify(tasks))
      return true
    } catch (error) {
      if (error.name === 'QuotaExceededError') {
        alert('保存容量が不足しています')
      } else {
        alert('保存に失敗しました')
      }
      return false
    }
  }

  // タスク追加
  function addTask(text) {
    if (!text || !text.trim()) {
      alert('タスクを入力してください')
      return false
    }

    tasks.push({
      id: Date.now(),
      text: text.trim(),
      done: false,
      createdAt: new Date().toISOString()
    })

    if (saveTasks()) {
      renderTasks()
      return true
    }
    return false
  }

  // レンダリング
  function renderTasks() {
    const list = document.getElementById('taskList')
    list.innerHTML = tasks.map(task => createTaskHTML(task)).join('')
  }

  // タスクHTML生成
  function createTaskHTML(task) {
    return `
      <li class="task-item ${task.done ? 'done' : ''}">
        <input type="checkbox"
               ${task.done ? 'checked' : ''}
               onchange="TodoApp.toggleTask(${task.id})"
               aria-label="タスク完了チェック">
        <span class="task-text">${escapeHTML(task.text)}</span>
        <button onclick="TodoApp.deleteTask(${task.id})"
                aria-label="タスク削除">
          削除
        </button>
      </li>
    `
  }

  // XSS対策
  function escapeHTML(str) {
    const div = document.createElement('div')
    div.textContent = str
    return div.innerHTML
  }

  // イベントリスナー
  function attachEventListeners() {
    const input = document.getElementById('taskInput')
    const addButton = document.getElementById('addButton')

    addButton.addEventListener('click', () => {
      if (addTask(input.value)) {
        input.value = ''
      }
    })

    // Enter キーで追加
    input.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') {
        addButton.click()
      }
    })
  }

  // 公開API
  return {
    init,
    addTask,
    toggleTask: (id) => {
      const task = tasks.find(t => t.id === id)
      if (task) {
        task.done = !task.done
        saveTasks()
        renderTasks()
      }
    },
    deleteTask: (id) => {
      tasks = tasks.filter(t => t.id !== id)
      saveTasks()
      renderTasks()
    }
  }
})()

// アプリ起動
document.addEventListener('DOMContentLoaded', () => {
  TodoApp.init()
})

function filterTasks(filter) {

    case 'all':
      return tasks
    case 'active':

    case 'completed':
      return tasks.filter(t => t.done)

  }

function sortByPriority(tasks) {

// 連続達成日数の計算
function calculateStreak(habit) {
  let streak = 0
  const today = new Date()

  for (let i = 0; i < habit.history.length; i++) {
    const date = new Date(habit.history[i].date)
    const daysDiff = Math.floor((today - date) / (1000 * 60 * 60 * 24))

    if (daysDiff === i && habit.history[i].done) {
      streak++
    } else {
      break
    }
  }

  return streak
}

  → WebアプリだけではUXに限界

  → オフライン対応が必要

❌ チーム開発が必要
  → コントリビューターが増えた
  → コードレビューの体制が必要
  → CI/CD自動化が必要

❌ ユーザーが増えてきた
  → 配布が必要（App Store等）
  → 品質保証が重要
  → サポート体制が必要

Step 1: 個人で作る（06章）
└─ Claude Webチャットのみ

├─ コード品質向上
├─ 機能拡張
└─ **特化型機能追加**（重要！）
    └─ データを自由に活用
    └─ 独自の価値創出
    └─ コミュニティの成長

Step 3: 本格開発へ（05章）
└─ 必要になったら
└─ Claude Code + Flutter
└─ マルチプラットフォーム展開

一人で完璧を目指さない
↓

特化型機能の重要性

BYOAの真価は、特化型機能にある

汎用アプリでは実現できない価値:

【育児アプリ】
記録 → アルバム自動生成
→ 祖父母と共有
→ 成長の可視化

【栽培アプリ】
栽培記録 → 販売用ストーリー
→ 付加価値の創出
→ 顧客との信頼構築

【家計簿アプリ】
収支記録 → 投資分析
→ 最適なポートフォリオ
→ ライフプラン設計

【読書アプリ】
読書記録 → 書評自動生成
→ コンテンツ化
→ 知識の体系化

データを自分で持つから、自由に活用できる 当事者だから、本当に必要な機能がわかる コミュニティがあるから、一緒に育てられる
